#include <stdio.h>
#include <math.h>

void ab(){
    printf("Neon");
}

void ba(){
    printf("Not Neon");
}

void res(int n){
    int m = pow(n,2), sum = 0;
    while(m!=0){
        int temp = m%10;
        sum += temp;
        m/=10;
    }
    (n==sum)?ab():ba();
}

int main() {
    int n;
    scanf("%d",&n);
    res(n);
    return 0;
}
