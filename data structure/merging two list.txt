#include <stdio.h>
#include <stdlib.h>
struct node{
    int data;
    struct node* next;
};
void insertEnd(struct node** head, int new_data){
    struct node* new_node = malloc(sizeof(struct node));
    new_node->data = new_data;
    new_node->next = NULL;
    if((*head)==NULL){
        *head = new_node;
        return;
    }
    struct node* temp = *head;
    while(temp->next!=NULL){
        temp = temp->next;
    }
    temp->next = new_node;
}

void display(struct node* head){
    while(head!=NULL){
        printf("%d ",head->data);
        head = head->next;
    }
}
int main(){
    struct node* head = NULL;
    insertEnd(&head,10);
    insertEnd(&head,20);
    insertEnd(&head,30);
    insertEnd(&head,40);

    struct node* element = NULL;
